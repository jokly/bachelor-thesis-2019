\textbf{Описание параметров}
\begin{itemize}
  \item user\_id - ID пользователя
  \item partner\_slug - идентификатор партнера (vl или dvhab)
  \item tag\_id - ID тега
  \item company\_id ID компании
\end{itemize}

\textbf{Рекомендация блюд}

\begin{itemize}
  \item Рекомендация из всего списка

  \apiurl{/api/items/all?user\_id=<user\_id>\&partner=<partner\_slug>}

  \item Рекомендации по тегу

  \apiurl{/api/items/by-tag/<tag\_id>?user\_id=<user\_id>\&partner=<partner\_slug>}

  \item Рекомендации по компании

  \apiurl{/api/items/by-company/<company\_id>?user\_id=<user\_id>\&partner=<partner\_slug>}
\end{itemize}

\textbf{Рекомендация компаний}

\begin{itemize}
  \item Рекомендации из всего списка:

  \apiurl{/api/companies/all?user\_id=<user\_id>\&partner=<partner\_slug>}

  \item Рекомендации по тегу:

  \apiurl{/api/companies/by-tag/<tag\_id>?user\_id=<user\_id>\&partner=<partner\_slug>}
\end{itemize}

\textbf{Рекомендация для тега}

\begin{itemize}
  \item Рекомендация для тега "Рекомендуем":

  \apiurl{/api/companies/all?user\_id=<user\_id>\&partner=<partner\_slug>}
\end{itemize}
